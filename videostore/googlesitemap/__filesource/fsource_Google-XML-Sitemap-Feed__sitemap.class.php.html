  178.     function DBPerform($table, $data, $action = 'insert', $parameters = '') {
    179. 

  179.         reset($data);
    180. 

  180.         if ($action == 'insert') {
    181. 

  181.           $query = 'INSERT INTO `' . $table . '` (';
    182. 

  182.           while (list($columns, ) = each($data)) {
    183. 

  183.             $query .= '`' . $columns . '`, ';
    184. 

  184.           }
    185. 

  185.           $query = substr($query, 0, -2) . ') values (';
    186. 

  186.           reset($data);
    187. 

  187.           while (list(, $value) = each($data)) {
    188. 

  188.             switch ((string)$value) {
    189. 

  189.               case 'now()':
    190. 

  190.                 $query .= 'now(), ';
    191. 

  191.                 break;
    192. 

  192.               case 'null':
    193. 

  193.                 $query .= 'null, ';
    194. 

  194.                 break;
    195. 

  195.               default:
    196. 

  196.                 $query .= "'" . $this->Slashes($value) . "', ";
    197. 

  197.                 break;
    198. 

  198.             }
    199. 

  199.           }
    200. 

  200.           $query = substr($query, 0, -2) . ')';
    201. 

  201.         } elseif ($action == 'update') {
    202. 

  202.           $query = 'UPDATE `' . $table . '` SET ';
